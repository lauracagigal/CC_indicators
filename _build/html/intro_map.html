import folium
import pandas as pd
import geopandas as gpd

lon_lat = [134.5, 5.5]

shp_f = './data/Palau_shp/pw_cz_pol_april2022.shp'
data = gpd.read_file(shp_f)

# Create a map centered at the first station's coordinates
map = folium.Map(location=[lon_lat[1], lon_lat[0]], zoom_start=7)

# Define the style function to change the color
def style_function(feature):
    return {
        'fillColor': 'crimson',  # Change the color here
        'color': 'crimson',
        'weight': .5,
        'fillOpacity': 0.2
    }

# Add the shape data to the map with the style function
folium.GeoJson(data, style_function=style_function).add_to(map)

# Display the map
map
